9pub(crate) fn bind(path: &Path) -> io::Result<net::UnixListener> {
10    let (sockaddr, socklen) = socket_addr(path)?;
11    let sockaddr = &sockaddr as *const libc::sockaddr_un as *const libc::sockaddr;
12
13    let fd = new_socket(libc::AF_UNIX, libc::SOCK_STREAM)?;
14    let socket = unsafe { net::UnixListener::from_raw_fd(fd) };
15    syscall!(bind(fd, sockaddr, socklen))?;
16    syscall!(listen(fd, 1024))?;
17
18    Ok(socket)
19}

21pub(crate) fn accept(listener: &net::UnixListener) -> io::Result<(UnixStream, SocketAddr)> {
22    let sockaddr = mem::MaybeUninit::<libc::sockaddr_un>::zeroed();
23
24    // This is safe to assume because a `libc::sockaddr_un` filled with `0`
25    // bytes is properly initialized.
26    //
27    // `0` is a valid value for `sockaddr_un::sun_family`; it is
28    // `libc::AF_UNSPEC`.
29    //
30    // `[0; 108]` is a valid value for `sockaddr_un::sun_path`; it begins an
31    // abstract path.
32    let mut sockaddr = unsafe { sockaddr.assume_init() };
33
34    sockaddr.sun_family = libc::AF_UNIX as libc::sa_family_t;
35    let mut socklen = mem::size_of_val(&sockaddr) as libc::socklen_t;
36
37    #[cfg(not(any(
38        target_os = "ios",
39        target_os = "macos",
40        target_os = "netbsd",
41        target_os = "redox",
42        // Android x86's seccomp profile forbids calls to `accept4(2)`
43        // See https://github.com/tokio-rs/mio/issues/1445 for details
44        all(target_arch = "x86", target_os = "android"),
45    )))]
46    let socket = {
47        let flags = libc::SOCK_NONBLOCK | libc::SOCK_CLOEXEC;
48        syscall!(accept4(
49            listener.as_raw_fd(),
50            &mut sockaddr as *mut libc::sockaddr_un as *mut libc::sockaddr,
51            &mut socklen,
52            flags
53        ))
54        .map(|socket| unsafe { net::UnixStream::from_raw_fd(socket) })
55    };
56
57    #[cfg(any(
58        target_os = "netbsd",
59        target_os = "redox",
60        target_os = "ios",
61        target_os = "macos",
62        all(target_arch = "x86", target_os = "android")
63    ))]
64    let socket = syscall!(accept(
65        listener.as_raw_fd(),
66        &mut sockaddr as *mut libc::sockaddr_un as *mut libc::sockaddr,
67        &mut socklen,
68    ))
69    .and_then(|socket| {
70        // Ensure the socket is closed if either of the `fcntl` calls
71        // error below.
72        let s = unsafe { net::UnixStream::from_raw_fd(socket) };
73        syscall!(fcntl(socket, libc::F_SETFD, libc::FD_CLOEXEC))?;
74
75        // See https://github.com/tokio-rs/mio/issues/1450
76        #[cfg(all(target_arch = "x86", target_os = "android"))]
77        syscall!(fcntl(socket, libc::F_SETFL, libc::O_NONBLOCK))?;
78
79        Ok(s)
80    });
81
82    socket
83        .map(UnixStream::from_std)
84        .map(|stream| (stream, SocketAddr::from_parts(sockaddr, socklen)))
85}
