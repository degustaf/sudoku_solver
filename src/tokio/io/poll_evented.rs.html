11cfg_io_driver! {
12    /// Associates an I/O resource that implements the [`std::io::Read`] and/or
13    /// [`std::io::Write`] traits with the reactor that drives it.
14    ///
15    /// `PollEvented` uses [`Registration`] internally to take a type that
16    /// implements [`mio::event::Source`] as well as [`std::io::Read`] and/or
17    /// [`std::io::Write`] and associate it with a reactor that will drive it.
18    ///
19    /// Once the [`mio::event::Source`] type is wrapped by `PollEvented`, it can be
20    /// used from within the future's execution model. As such, the
21    /// `PollEvented` type provides [`AsyncRead`] and [`AsyncWrite`]
22    /// implementations using the underlying I/O resource as well as readiness
23    /// events provided by the reactor.
24    ///
25    /// **Note**: While `PollEvented` is `Sync` (if the underlying I/O type is
26    /// `Sync`), the caller must ensure that there are at most two tasks that
27    /// use a `PollEvented` instance concurrently. One for reading and one for
28    /// writing. While violating this requirement is "safe" from a Rust memory
29    /// model point of view, it will result in unexpected behavior in the form
30    /// of lost notifications and tasks hanging.
31    ///
32    /// ## Readiness events
33    ///
34    /// Besides just providing [`AsyncRead`] and [`AsyncWrite`] implementations,
35    /// this type also supports access to the underlying readiness event stream.
36    /// While similar in function to what [`Registration`] provides, the
37    /// semantics are a bit different.
38    ///
39    /// Two functions are provided to access the readiness events:
40    /// [`poll_read_ready`] and [`poll_write_ready`]. These functions return the
41    /// current readiness state of the `PollEvented` instance. If
42    /// [`poll_read_ready`] indicates read readiness, immediately calling
43    /// [`poll_read_ready`] again will also indicate read readiness.
44    ///
45    /// When the operation is attempted and is unable to succeed due to the I/O
46    /// resource not being ready, the caller must call [`clear_readiness`].
47    /// This clears the readiness state until a new readiness event is received.
48    ///
49    /// This allows the caller to implement additional functions. For example,
50    /// [`TcpListener`] implements poll_accept by using [`poll_read_ready`] and
51    /// [`clear_readiness`].
52    ///
53    /// ## Platform-specific events
54    ///
55    /// `PollEvented` also allows receiving platform-specific `mio::Ready` events.
56    /// These events are included as part of the read readiness event stream. The
57    /// write readiness event stream is only for `Ready::writable()` events.
58    ///
59    /// [`AsyncRead`]: crate::io::AsyncRead
60    /// [`AsyncWrite`]: crate::io::AsyncWrite
61    /// [`TcpListener`]: crate::net::TcpListener
62    /// [`clear_readiness`]: Registration::clear_readiness
63    /// [`poll_read_ready`]: Registration::poll_read_ready
64    /// [`poll_write_ready`]: Registration::poll_write_ready
65    pub(crate) struct PollEvented<E: Source> {
66        io: Option<E>,
67        registration: Registration,
68    }
69}

73impl<E: Source> PollEvented<E> {
74    /// Creates a new `PollEvented` associated with the default reactor.
75    ///
76    /// # Panics
77    ///
78    /// This function panics if thread-local runtime is not set.
79    ///
80    /// The runtime is usually set implicitly when this function is called
81    /// from a future driven by a tokio runtime, otherwise runtime can be set
82    /// explicitly with [`Runtime::enter`](crate::runtime::Runtime::enter) function.
83    #[track_caller]
84    #[cfg_attr(feature = "signal", allow(unused))]
85    pub(crate) fn new(io: E) -> io::Result<Self> {
86        PollEvented::new_with_interest(io, Interest::READABLE | Interest::WRITABLE)
87    }
88
89    /// Creates a new `PollEvented` associated with the default reactor, for
90    /// specific `Interest` state. `new_with_interest` should be used over `new`
91    /// when you need control over the readiness state, such as when a file
92    /// descriptor only allows reads. This does not add `hup` or `error` so if
93    /// you are interested in those states, you will need to add them to the
94    /// readiness state passed to this function.
95    ///
96    /// # Panics
97    ///
98    /// This function panics if thread-local runtime is not set.
99    ///
100    /// The runtime is usually set implicitly when this function is called from
101    /// a future driven by a tokio runtime, otherwise runtime can be set
102    /// explicitly with [`Runtime::enter`](crate::runtime::Runtime::enter)
103    /// function.
104    #[track_caller]
105    #[cfg_attr(feature = "signal", allow(unused))]
106    pub(crate) fn new_with_interest(io: E, interest: Interest) -> io::Result<Self> {
107        Self::new_with_interest_and_handle(io, interest, scheduler::Handle::current())
108    }
109
110    #[track_caller]
111    pub(crate) fn new_with_interest_and_handle(
112        mut io: E,
113        interest: Interest,
114        handle: scheduler::Handle,
115    ) -> io::Result<Self> {
116        let registration = Registration::new_with_interest_and_handle(&mut io, interest, handle)?;
117        Ok(Self {
118            io: Some(io),
119            registration,
120        })
121    }
122
123    /// Returns a reference to the registration.
124    #[cfg(any(feature = "net"))]
125    pub(crate) fn registration(&self) -> &Registration {
126        &self.registration
127    }
128
129    /// Deregisters the inner io from the registration and returns a Result containing the inner io.
130    #[cfg(any(feature = "net", feature = "process"))]
131    pub(crate) fn into_inner(mut self) -> io::Result<E> {
132        let mut inner = self.io.take().unwrap(); // As io shouldn't ever be None, just unwrap here.
133        self.registration.deregister(&mut inner)?;
134        Ok(inner)
135    }
136}

141    use crate::io::ReadBuf;
142    use std::task::{Context, Poll};
143
144    impl<E: Source> PollEvented<E> {
145        // Safety: The caller must ensure that `E` can read into uninitialized memory
146        pub(crate) unsafe fn poll_read<'a>(
147            &'a self,
148            cx: &mut Context<'_>,
149            buf: &mut ReadBuf<'_>,
150        ) -> Poll<io::Result<()>>
151        where
152            &'a E: io::Read + 'a,
153        {
154            use std::io::Read;
155
156            loop {
157                let evt = ready!(self.registration.poll_read_ready(cx))?;
158
159                let b = &mut *(buf.unfilled_mut() as *mut [std::mem::MaybeUninit<u8>] as *mut [u8]);
160                let len = b.len();
161
162                match self.io.as_ref().unwrap().read(b) {
163                    Ok(n) => {
164                        // if we read a partially full buffer, this is sufficient on unix to show
165                        // that the socket buffer has been drained
166                        if n > 0 && (!cfg!(windows) && n < len) {
167                            self.registration.clear_readiness(evt);
168                        }
169
170                        // Safety: We trust `TcpStream::read` to have filled up `n` bytes in the
171                        // buffer.
172                        buf.assume_init(n);
173                        buf.advance(n);
174                        return Poll::Ready(Ok(()));
175                    },
176                    Err(e) if e.kind() == io::ErrorKind::WouldBlock => {
177                        self.registration.clear_readiness(evt);
178                    }
179                    Err(e) => return Poll::Ready(Err(e)),
180                }
181            }
182        }
183
184        pub(crate) fn poll_write<'a>(&'a self, cx: &mut Context<'_>, buf: &[u8]) -> Poll<io::Result<usize>>
185        where
186            &'a E: io::Write + 'a,
187        {
188            use std::io::Write;
189
190            loop {
191                let evt = ready!(self.registration.poll_write_ready(cx))?;
192
193                match self.io.as_ref().unwrap().write(buf) {
194                    Ok(n) => {
195                        // if we write only part of our buffer, this is sufficient on unix to show
196                        // that the socket buffer is full
197                        if n > 0 && (!cfg!(windows) && n < buf.len()) {
198                            self.registration.clear_readiness(evt);
199                        }
200
201                        return Poll::Ready(Ok(n));
202                    },
203                    Err(e) if e.kind() == io::ErrorKind::WouldBlock => {
204                        self.registration.clear_readiness(evt);
205                    }
206                    Err(e) => return Poll::Ready(Err(e)),
207                }
208            }
209        }
210
211        #[cfg(any(feature = "net", feature = "process"))]
212        pub(crate) fn poll_write_vectored<'a>(
213            &'a self,
214            cx: &mut Context<'_>,
215            bufs: &[io::IoSlice<'_>],
216        ) -> Poll<io::Result<usize>>
217        where
218            &'a E: io::Write + 'a,
219        {
220            use std::io::Write;
221            self.registration.poll_write_io(cx, || self.io.as_ref().unwrap().write_vectored(bufs))
222        }
223    }
224}

244impl<E: Source> Drop for PollEvented<E> {
245    fn drop(&mut self) {
246        if let Some(mut io) = self.io.take() {
247            // Ignore errors
248            let _ = self.registration.deregister(&mut io);
249        }
250    }
