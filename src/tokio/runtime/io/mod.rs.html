82// TODO: Don't use a fake token. Instead, reserve a slot entry for the wakeup
83// token.
84const TOKEN_WAKEUP: mio::Token = mio::Token(1 << 31);
85const TOKEN_SIGNAL: mio::Token = mio::Token(1 + (1 << 31));
86
87const ADDRESS: bit::Pack = bit::Pack::least_significant(24);
88
89// Packs the generation value in the `readiness` field.
90//
91// The generation prevents a race condition where a slab slot is reused for a
92// new socket while the I/O driver is about to apply a readiness event. The
93// generation value is checked when setting new readiness. If the generation do
94// not match, then the readiness event is discarded.
95const GENERATION: bit::Pack = ADDRESS.then(7);

105impl Driver {
106    /// Creates a new event loop, returning any error that happened during the
107    /// creation.
108    pub(crate) fn new(nevents: usize) -> io::Result<(Driver, Handle)> {
109        let poll = mio::Poll::new()?;
110        #[cfg(not(tokio_wasi))]
111        let waker = mio::Waker::new(poll.registry(), TOKEN_WAKEUP)?;
112        let registry = poll.registry().try_clone()?;
113
114        let slab = Slab::new();
115        let allocator = slab.allocator();
116
117        let driver = Driver {
118            tick: 0,
119            signal_ready: false,
120            events: mio::Events::with_capacity(nevents),
121            poll,
122            resources: slab,
123        };
124
125        let handle = Handle {
126            registry,
127            io_dispatch: RwLock::new(IoDispatcher::new(allocator)),
128            #[cfg(not(tokio_wasi))]
129            waker,
130            metrics: IoDriverMetrics::default(),
131        };
132
133        Ok((driver, handle))
134    }
135
136    pub(crate) fn park(&mut self, rt_handle: &driver::Handle) {
137        let handle = rt_handle.io();
138        self.turn(handle, None);
139    }
140
141    pub(crate) fn park_timeout(&mut self, rt_handle: &driver::Handle, duration: Duration) {
142        let handle = rt_handle.io();
143        self.turn(handle, Some(duration));
144    }
145
146    pub(crate) fn shutdown(&mut self, rt_handle: &driver::Handle) {
147        let handle = rt_handle.io();
148
149        if handle.shutdown() {
150            self.resources.for_each(|io| {
151                // If a task is waiting on the I/O resource, notify it that the
152                // runtime is being shutdown. And shutdown will clear all wakers.
153                io.shutdown();
154            });
155        }
156    }
157
158    fn turn(&mut self, handle: &Handle, max_wait: Option<Duration>) {
159        // How often to call `compact()` on the resource slab
160        const COMPACT_INTERVAL: u8 = 255;
161
162        self.tick = self.tick.wrapping_add(1);
163
164        if self.tick == COMPACT_INTERVAL {
165            self.resources.compact()
166        }
167
168        let events = &mut self.events;
169
170        // Block waiting for an event to happen, peeling out how many events
171        // happened.
172        match self.poll.poll(events, max_wait) {
173            Ok(_) => {}
174            Err(ref e) if e.kind() == io::ErrorKind::Interrupted => {}
175            #[cfg(tokio_wasi)]
176            Err(e) if e.kind() == io::ErrorKind::InvalidInput => {
177                // In case of wasm32_wasi this error happens, when trying to poll without subscriptions
178                // just return from the park, as there would be nothing, which wakes us up.
179            }
180            Err(e) => panic!("unexpected error when polling the I/O driver: {:?}", e),
181        }
182
183        // Process all the events that came in, dispatching appropriately
184        let mut ready_count = 0;
185        for event in events.iter() {
186            let token = event.token();
187
188            if token == TOKEN_WAKEUP {
189                // Nothing to do, the event is used to unblock the I/O driver
190            } else if token == TOKEN_SIGNAL {
191                self.signal_ready = true;
192            } else {
193                Self::dispatch(
194                    &mut self.resources,
195                    self.tick,
196                    token,
197                    Ready::from_mio(event),
198                );
199                ready_count += 1;
200            }
201        }
202
203        handle.metrics.incr_ready_count_by(ready_count);
204    }
205
206    fn dispatch(resources: &mut Slab<ScheduledIo>, tick: u8, token: mio::Token, ready: Ready) {
207        let addr = slab::Address::from_usize(ADDRESS.unpack(token.0));
208
209        let io = match resources.get(addr) {
210            Some(io) => io,
211            None => return,
212        };
213
214        let res = io.set_readiness(Some(token.0), Tick::Set(tick), |curr| curr | ready);
215
216        if res.is_err() {
217            // token no longer valid!
218            return;
219        }
220
221        io.wake(ready);
222    }
223}

231impl Handle {
232    /// Forces a reactor blocked in a call to `turn` to wakeup, or otherwise
233    /// makes the next call to `turn` return immediately.
234    ///
235    /// This method is intended to be used in situations where a notification
236    /// needs to otherwise be sent to the main reactor. If the reactor is
237    /// currently blocked inside of `turn` then it will wake up and soon return
238    /// after this method has been called. If the reactor is not currently
239    /// blocked in `turn`, then the next call to `turn` will not block and
240    /// return immediately.
241    pub(crate) fn unpark(&self) {
242        #[cfg(not(tokio_wasi))]
243        self.waker.wake().expect("failed to wake I/O driver");
244    }
245
246    /// Registers an I/O resource with the reactor for a given `mio::Ready` state.
247    ///
248    /// The registration token is returned.
249    pub(super) fn add_source(
250        &self,
251        source: &mut impl mio::event::Source,
252        interest: Interest,
253    ) -> io::Result<slab::Ref<ScheduledIo>> {
254        let (address, shared) = self.allocate()?;
255
256        let token = GENERATION.pack(shared.generation(), ADDRESS.pack(address.as_usize(), 0));
257
258        self.registry
259            .register(source, mio::Token(token), interest.to_mio())?;
260
261        self.metrics.incr_fd_count();
262
263        Ok(shared)
264    }
265
266    /// Deregisters an I/O resource from the reactor.
267    pub(super) fn deregister_source(&self, source: &mut impl mio::event::Source) -> io::Result<()> {
268        self.registry.deregister(source)?;
269
270        self.metrics.dec_fd_count();
271
272        Ok(())
273    }
274
275    /// shutdown the dispatcher.
276    fn shutdown(&self) -> bool {
277        let mut io = self.io_dispatch.write().unwrap();
278        if io.is_shutdown {
279            return false;
280        }
281        io.is_shutdown = true;
282        true
283    }
284
285    fn allocate(&self) -> io::Result<(slab::Address, slab::Ref<ScheduledIo>)> {
286        let io = self.io_dispatch.read().unwrap();
287        if io.is_shutdown {
288            return Err(io::Error::new(
289                io::ErrorKind::Other,
290                crate::util::error::RUNTIME_SHUTTING_DOWN_ERROR,
291            ));
292        }
293        io.allocator.allocate().ok_or_else(|| {
294            io::Error::new(
295                io::ErrorKind::Other,
296                "reactor at max registered I/O resources",
297            )
298        })
299    }
300}

319impl Direction {
320    pub(super) fn mask(self) -> Ready {
321        match self {
