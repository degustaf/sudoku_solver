9/// Builds Tokio Runtime with custom configuration values.
10///
11/// Methods can be chained in order to set the configuration values. The
12/// Runtime is constructed by calling [`build`].
13///
14/// New instances of `Builder` are obtained via [`Builder::new_multi_thread`]
15/// or [`Builder::new_current_thread`].
16///
17/// See function level documentation for details on the various configuration
18/// settings.
19///
20/// [`build`]: method@Self::build
21/// [`Builder::new_multi_thread`]: method@Self::new_multi_thread
22/// [`Builder::new_current_thread`]: method@Self::new_current_thread
23///
24/// # Examples
25///
26/// ```
27/// use tokio::runtime::Builder;
28///
29/// fn main() {
30///     // build runtime
31///     let runtime = Builder::new_multi_thread()
32///         .worker_threads(4)
33///         .thread_name("my-custom-name")
34///         .thread_stack_size(3 * 1024 * 1024)
35///         .build()
36///         .unwrap();
37///
38///     // use runtime ...
39/// }
40/// ```

47    nevents: usize,

54

57    /// Only used when not using the current-thread executor.
58    worker_threads: Option<usize>,

60    /// Cap on thread usage.
61    max_blocking_threads: usize,

102cfg_unstable! {
103    /// How the runtime should respond to unhandled panics.
104    ///
105    /// Instances of `UnhandledPanic` are passed to `Builder::unhandled_panic`
106    /// to configure the runtime behavior when a spawned task panics.
107    ///
108    /// See [`Builder::unhandled_panic`] for more details.
109    #[derive(Debug, Clone)]
110    #[non_exhaustive]
111    pub enum UnhandledPanic {
112        /// The runtime should ignore panics on spawned tasks.
113        ///
114        /// The panic is forwarded to the task's [`JoinHandle`] and all spawned
115        /// tasks continue running normally.
116        ///
117        /// This is the default behavior.
118        ///
119        /// # Examples
120        ///
121        /// ```
122        /// use tokio::runtime::{self, UnhandledPanic};
123        ///
124        /// # pub fn main() {
125        /// let rt = runtime::Builder::new_current_thread()
126        ///     .unhandled_panic(UnhandledPanic::Ignore)
127        ///     .build()
128        ///     .unwrap();
129        ///
130        /// let task1 = rt.spawn(async { panic!("boom"); });
131        /// let task2 = rt.spawn(async {
132        ///     // This task completes normally
133        ///     "done"
134        /// });
135        ///
136        /// rt.block_on(async {
137        ///     // The panic on the first task is forwarded to the `JoinHandle`
138        ///     assert!(task1.await.is_err());
139        ///
140        ///     // The second task completes normally
141        ///     assert!(task2.await.is_ok());
142        /// })
143        /// # }
144        /// ```
145        ///
146        /// [`JoinHandle`]: struct@crate::task::JoinHandle
147        Ignore,
148
149        /// The runtime should immediately shutdown if a spawned task panics.
150        ///
151        /// The runtime will immediately shutdown even if the panicked task's
152        /// [`JoinHandle`] is still available. All further spawned tasks will be
153        /// immediately dropped and call to [`Runtime::block_on`] will panic.
154        ///
155        /// # Examples
156        ///
157        /// ```should_panic
158        /// use tokio::runtime::{self, UnhandledPanic};
159        ///
160        /// # pub fn main() {
161        /// let rt = runtime::Builder::new_current_thread()
162        ///     .unhandled_panic(UnhandledPanic::ShutdownRuntime)
163        ///     .build()
164        ///     .unwrap();
165        ///
166        /// rt.spawn(async { panic!("boom"); });
167        /// rt.spawn(async {
168        ///     // This task never completes.
169        /// });
170        ///
171        /// rt.block_on(async {
172        ///     // Do some work
173        /// # loop { tokio::task::yield_now().await; }
174        /// })
175        /// # }
176        /// ```
177        ///
178        /// [`JoinHandle`]: struct@crate::task::JoinHandle
179        ShutdownRuntime,
180    }
181}

304    /// Sets the number of worker threads the `Runtime` will use.
305    ///
306    /// This can be any number above 0 though it is advised to keep this value
307    /// on the smaller side.
308    ///
309    /// This will override the value read from environment variable `TOKIO_WORKER_THREADS`.
310    ///
311    /// # Default
312    ///
313    /// The default value is the number of cores available to the system.
314    ///
315    /// When using the `current_thread` runtime this method has no effect.
316    ///
317    /// # Examples
318    ///
319    /// ## Multi threaded runtime with 4 threads
320    ///
321    /// ```
322    /// use tokio::runtime;
323    ///
324    /// // This will spawn a work-stealing runtime with 4 worker threads.
325    /// let rt = runtime::Builder::new_multi_thread()
326    ///     .worker_threads(4)
327    ///     .build()
328    ///     .unwrap();
329    ///
330    /// rt.spawn(async move {});
331    /// ```
332    ///
333    /// ## Current thread runtime (will only run on the current thread via `Runtime::block_on`)
334    ///
335    /// ```
336    /// use tokio::runtime;
337    ///
338    /// // Create a runtime that _must_ be driven from a call
339    /// // to `Runtime::block_on`.
340    /// let rt = runtime::Builder::new_current_thread()
341    ///     .build()
342    ///     .unwrap();
343    ///
344    /// // This will run the runtime and future on the current thread
345    /// rt.block_on(async move {});
346    /// ```
347    ///
348    /// # Panics
349    ///
350    /// This will panic if `val` is not larger than `0`.

510    /// Executes function `f` just before a thread is parked (goes idle).
511    /// `f` is called within the Tokio context, so functions like [`tokio::spawn`](crate::spawn)
512    /// can be called, and may result in this thread being unparked immediately.
513    ///
514    /// This can be used to start work only when the executor is idle, or for bookkeeping
515    /// and monitoring purposes.
516    ///
517    /// Note: There can only be one park callback for a runtime; calling this function
518    /// more than once replaces the last callback defined, rather than adding to it.
519    ///
520    /// # Examples
521    ///
522    /// ## Multithreaded executor
523    /// ```
524    /// # use std::sync::Arc;
525    /// # use std::sync::atomic::{AtomicBool, Ordering};
526    /// # use tokio::runtime;
527    /// # use tokio::sync::Barrier;
528    /// # pub fn main() {
529    /// let once = AtomicBool::new(true);
530    /// let barrier = Arc::new(Barrier::new(2));
531    ///
532    /// let runtime = runtime::Builder::new_multi_thread()
533    ///     .worker_threads(1)
534    ///     .on_thread_park({
535    ///         let barrier = barrier.clone();
536    ///         move || {
537    ///             let barrier = barrier.clone();
538    ///             if once.swap(false, Ordering::Relaxed) {
539    ///                 tokio::spawn(async move { barrier.wait().await; });
540    ///            }
541    ///         }
542    ///     })
543    ///     .build()
544    ///     .unwrap();
545    ///
546    /// runtime.block_on(async {
547    ///    barrier.wait().await;
548    /// })
549    /// # }
550    /// ```
551    /// ## Current thread executor
552    /// ```
553    /// # use std::sync::Arc;
554    /// # use std::sync::atomic::{AtomicBool, Ordering};
555    /// # use tokio::runtime;
556    /// # use tokio::sync::Barrier;
557    /// # pub fn main() {
558    /// let once = AtomicBool::new(true);
559    /// let barrier = Arc::new(Barrier::new(2));
560    ///
561    /// let runtime = runtime::Builder::new_current_thread()
562    ///     .on_thread_park({
563    ///         let barrier = barrier.clone();
564    ///         move || {
565    ///             let barrier = barrier.clone();
566    ///             if once.swap(false, Ordering::Relaxed) {
567    ///                 tokio::spawn(async move { barrier.wait().await; });
568    ///            }
569    ///         }
570    ///     })
571    ///     .build()
572    ///     .unwrap();
573    ///
574    /// runtime.block_on(async {
575    ///    barrier.wait().await;
576    /// })
577    /// # }
578    /// ```
579    #[cfg(not(loom))]
580    pub fn on_thread_park<F>(&mut self, f: F) -> &mut Self
581    where

681    /// Sets the number of scheduler ticks after which the scheduler will poll the global
682    /// task queue.
683    ///
684    /// A scheduler "tick" roughly corresponds to one `poll` invocation on a task.
685    ///
686    /// By default the global queue interval is:
687    ///
688    /// * `31` for the current-thread scheduler.
689    /// * `61` for the multithreaded scheduler.
690    ///
691    /// Schedulers have a local queue of already-claimed tasks, and a global queue of incoming
692    /// tasks. Setting the interval to a smaller value increases the fairness of the scheduler,
693    /// at the cost of more synchronization overhead. That can be beneficial for prioritizing
694    /// getting started on new work, especially if tasks frequently yield rather than complete
695    /// or await on further I/O. Conversely, a higher value prioritizes existing work, and
696    /// is a good choice when most tasks quickly complete polling.
697    ///
698    /// # Examples
699    ///
700    /// ```
701    /// # use tokio::runtime;
702    /// # pub fn main() {
703    /// let rt = runtime::Builder::new_multi_thread()
704    ///     .global_queue_interval(31)
705    ///     .build();
706    /// # }
707    /// ```
708    pub fn global_queue_interval(&mut self, val: u32) -> &mut Self {
709        self.global_queue_interval = val;
710        self
711    }

713    /// Sets the number of scheduler ticks after which the scheduler will poll for
714    /// external events (timers, I/O, and so on).
715    ///
716    /// A scheduler "tick" roughly corresponds to one `poll` invocation on a task.
717    ///
718    /// By default, the event interval is `61` for all scheduler types.
719    ///
720    /// Setting the event interval determines the effective "priority" of delivering
721    /// these external events (which may wake up additional tasks), compared to
722    /// executing tasks that are currently ready to run. A smaller value is useful
723    /// when tasks frequently spend a long time in polling, or frequently yield,
724    /// which can result in overly long delays picking up I/O events. Conversely,
725    /// picking up new events requires extra synchronization and syscall overhead,
726    /// so if tasks generally complete their polling quickly, a higher event interval
727    /// will minimize that overhead while still keeping the scheduler responsive to
728    /// events.
729    ///
730    /// # Examples
731    ///
732    /// ```
733    /// # use tokio::runtime;
734    /// # pub fn main() {
735    /// let rt = runtime::Builder::new_multi_thread()
736    ///     .event_interval(31)
737    ///     .build();
738    /// # }
739    /// ```
740    pub fn event_interval(&mut self, val: u32) -> &mut Self {
741        self.event_interval = val;
742        self
743    }

745    cfg_unstable! {
746        /// Configure how the runtime responds to an unhandled panic on a
747        /// spawned task.
748        ///
749        /// By default, an unhandled panic (i.e. a panic not caught by
750        /// [`std::panic::catch_unwind`]) has no impact on the runtime's
751        /// execution. The panic is error value is forwarded to the task's
752        /// [`JoinHandle`] and all other spawned tasks continue running.
753        ///
754        /// The `unhandled_panic` option enables configuring this behavior.
755        ///
756        /// * `UnhandledPanic::Ignore` is the default behavior. Panics on
757        ///   spawned tasks have no impact on the runtime's execution.
758        /// * `UnhandledPanic::ShutdownRuntime` will force the runtime to
759        ///   shutdown immediately when a spawned task panics even if that
760        ///   task's `JoinHandle` has not been dropped. All other spawned tasks
761        ///   will immediately terminate and further calls to
762        ///   [`Runtime::block_on`] will panic.
763        ///
764        /// # Unstable
765        ///
766        /// This option is currently unstable and its implementation is
767        /// incomplete. The API may change or be removed in the future. See
768        /// tokio-rs/tokio#4516 for more details.
769        ///
770        /// # Examples
771        ///
772        /// The following demonstrates a runtime configured to shutdown on
773        /// panic. The first spawned task panics and results in the runtime
774        /// shutting down. The second spawned task never has a chance to
775        /// execute. The call to `block_on` will panic due to the runtime being
776        /// forcibly shutdown.
777        ///
778        /// ```should_panic
779        /// use tokio::runtime::{self, UnhandledPanic};
780        ///
781        /// # pub fn main() {
782        /// let rt = runtime::Builder::new_current_thread()
783        ///     .unhandled_panic(UnhandledPanic::ShutdownRuntime)
784        ///     .build()
785        ///     .unwrap();
786        ///
787        /// rt.spawn(async { panic!("boom"); });
788        /// rt.spawn(async {
789        ///     // This task never completes.
790        /// });
791        ///
792        /// rt.block_on(async {
793        ///     // Do some work
794        /// # loop { tokio::task::yield_now().await; }
795        /// })
796        /// # }
797        /// ```
798        ///
799        /// [`JoinHandle`]: struct@crate::task::JoinHandle
800        pub fn unhandled_panic(&mut self, behavior: UnhandledPanic) -> &mut Self {
801            self.unhandled_panic = behavior;
802            self
803        }

805        /// Disables the LIFO task scheduler heuristic.
806        ///
807        /// The multi-threaded scheduler includes a heuristic for optimizing
808        /// message-passing patterns. This heuristic results in the **last**
809        /// scheduled task being polled first.
810        ///
811        /// To implement this heuristic, each worker thread has a slot which
812        /// holds the task that should be polled next. However, this slot cannot
813        /// be stolen by other worker threads, which can result in lower total
814        /// throughput when tasks tend to have longer poll times.
815        ///
816        /// This configuration option will disable this heuristic resulting in
817        /// all scheduled tasks being pushed into the worker-local queue, which
818        /// is stealable.
819        ///
820        /// Consider trying this option when the task "scheduled" time is high
821        /// but the runtime is underutilized. Use tokio-rs/tokio-metrics to
822        /// collect this data.
823        ///
824        /// # Unstable
825        ///
826        /// This configuration option is considered a workaround for the LIFO
827        /// slot not being stealable. When the slot becomes stealable, we will
828        /// revisit whether or not this option is necessary. See
829        /// tokio-rs/tokio#4941.
830        ///
831        /// # Examples
832        ///
833        /// ```
834        /// use tokio::runtime;
835        ///
836        /// let rt = runtime::Builder::new_multi_thread()
837        ///     .disable_lifo_slot()
838        ///     .build()
839        ///     .unwrap();
840        /// ```
841        pub fn disable_lifo_slot(&mut self) -> &mut Self {
842            self.disable_lifo_slot = true;
843            self
844        }

846        /// Specifies the random number generation seed to use within all
847        /// threads associated with the runtime being built.
848        ///
849        /// This option is intended to make certain parts of the runtime
850        /// deterministic (e.g. the [`tokio::select!`] macro). In the case of
851        /// [`tokio::select!`] it will ensure that the order that branches are
852        /// polled is deterministic.
853        ///
854        /// In addition to the code specifying `rng_seed` and interacting with
855        /// the runtime, the internals of Tokio and the Rust compiler may affect
856        /// the sequences of random numbers. In order to ensure repeatable
857        /// results, the version of Tokio, the versions of all other
858        /// dependencies that interact with Tokio, and the Rust compiler version
859        /// should also all remain constant.
860        ///
861        /// # Examples
862        ///
863        /// ```
864        /// # use tokio::runtime::{self, RngSeed};
865        /// # pub fn main() {
866        /// let seed = RngSeed::from_bytes(b"place your seed here");
867        /// let rt = runtime::Builder::new_current_thread()
868        ///     .rng_seed(seed)
869        ///     .build();
870        /// # }
871        /// ```
872        ///
873        /// [`tokio::select!`]: crate::select
874        pub fn rng_seed(&mut self, seed: RngSeed) -> &mut Self {
875            self.seed_generator = RngSeedGenerator::new(seed);
876            self
877        }
878    }

880    fn build_current_thread_runtime(&mut self) -> io::Result<Runtime> {
881        use crate::runtime::scheduler::{self, CurrentThread};
882        use crate::runtime::{runtime::Scheduler, Config};
883
884        let (driver, driver_handle) = driver::Driver::new(self.get_cfg())?;
885
886        // Blocking pool
887        let blocking_pool = blocking::create_blocking_pool(self, self.max_blocking_threads);
888        let blocking_spawner = blocking_pool.spawner().clone();
889
890        // Generate a rng seed for this runtime.
891        let seed_generator_1 = self.seed_generator.next_generator();
892        let seed_generator_2 = self.seed_generator.next_generator();
893
894        // And now put a single-threaded scheduler on top of the timer. When
895        // there are no futures ready to do something, it'll let the timer or
896        // the reactor to generate some new stimuli for the futures to continue
897        // in their life.
898        let (scheduler, handle) = CurrentThread::new(
899            driver,
900            driver_handle,
901            blocking_spawner,
902            seed_generator_2,
903            Config {
904                before_park: self.before_park.clone(),
905                after_unpark: self.after_unpark.clone(),
906                global_queue_interval: self.global_queue_interval,
907                event_interval: self.event_interval,
908                #[cfg(tokio_unstable)]
909                unhandled_panic: self.unhandled_panic.clone(),
910                disable_lifo_slot: self.disable_lifo_slot,
911                seed_generator: seed_generator_1,
912            },
913        );
914
915        let handle = Handle {
916            inner: scheduler::Handle::CurrentThread(handle),
917        };
918
919        Ok(Runtime::from_parts(
920            Scheduler::CurrentThread(scheduler),
921            handle,
922            blocking_pool,
923        ))
924    }

1018cfg_rt_multi_thread! {
1019    impl Builder {
1020        fn build_threaded_runtime(&mut self) -> io::Result<Runtime> {
1021            use crate::loom::sys::num_cpus;
1022            use crate::runtime::{Config, runtime::Scheduler};
1023            use crate::runtime::scheduler::{self, MultiThread};
1024
1025            let core_threads = self.worker_threads.unwrap_or_else(num_cpus);
1026
1027            let (driver, driver_handle) = driver::Driver::new(self.get_cfg())?;
1028
1029            // Create the blocking pool
1030            let blocking_pool =
1031                blocking::create_blocking_pool(self, self.max_blocking_threads + core_threads);
1032            let blocking_spawner = blocking_pool.spawner().clone();
1033
1034            // Generate a rng seed for this runtime.
1035            let seed_generator_1 = self.seed_generator.next_generator();
1036            let seed_generator_2 = self.seed_generator.next_generator();
1037
1038            let (scheduler, handle, launch) = MultiThread::new(
1039                core_threads,
1040                driver,
1041                driver_handle,
1042                blocking_spawner,
1043                seed_generator_2,
1044                Config {
1045                    before_park: self.before_park.clone(),
1046                    after_unpark: self.after_unpark.clone(),
1047                    global_queue_interval: self.global_queue_interval,
1048                    event_interval: self.event_interval,
1049                    #[cfg(tokio_unstable)]
1050                    unhandled_panic: self.unhandled_panic.clone(),
1051                    disable_lifo_slot: self.disable_lifo_slot,
1052                    seed_generator: seed_generator_1,
1053                },
1054            );
1055
1056            let handle = Handle { inner: scheduler::Handle::MultiThread(handle) };
1057
1058            // Spawn the thread pool workers
1059            let _enter = handle.enter();
1060            launch.launch();
1061
1062            Ok(Runtime::from_parts(Scheduler::MultiThread(scheduler), handle, blocking_pool))
1063        }
1064    }
