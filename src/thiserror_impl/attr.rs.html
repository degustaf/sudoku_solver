47pub fn get(input: &[Attribute]) -> Result<Attrs> {
48    let mut attrs = Attrs {
49        display: None,
50        source: None,
51        backtrace: None,
52        from: None,
53        transparent: None,
54    };
55
56    for attr in input {
57        if attr.path().is_ident("error") {
58            parse_error_attribute(&mut attrs, attr)?;
59        } else if attr.path().is_ident("source") {
60            require_empty_attribute(attr)?;
61            if attrs.source.is_some() {
62                return Err(Error::new_spanned(attr, "duplicate #[source] attribute"));
63            }
64            attrs.source = Some(attr);
65        } else if attr.path().is_ident("backtrace") {
66            require_empty_attribute(attr)?;
67            if attrs.backtrace.is_some() {
68                return Err(Error::new_spanned(attr, "duplicate #[backtrace] attribute"));
69            }
70            attrs.backtrace = Some(attr);
71        } else if attr.path().is_ident("from") {
72            match attr.meta {
73                Meta::Path(_) => {}
74                Meta::List(_) | Meta::NameValue(_) => {
75                    // Assume this is meant for derive_more crate or something.
76                    continue;
77                }
78            }
79            if attrs.from.is_some() {
80                return Err(Error::new_spanned(attr, "duplicate #[from] attribute"));
81            }
82            attrs.from = Some(attr);
83        }
84    }
85
86    Ok(attrs)
87}

89fn parse_error_attribute<'a>(attrs: &mut Attrs<'a>, attr: &'a Attribute) -> Result<()> {
90    syn::custom_keyword!(transparent);
91
92    attr.parse_args_with(|input: ParseStream| {
93        if let Some(kw) = input.parse::<Option<transparent>>()? {
94            if attrs.transparent.is_some() {
95                return Err(Error::new_spanned(
96                    attr,
97                    "duplicate #[error(transparent)] attribute",
98                ));
99            }
100            attrs.transparent = Some(Transparent {
101                original: attr,
102                span: kw.span,
103            });
104            return Ok(());
105        }
106
107        let display = Display {
108            original: attr,
109            fmt: input.parse()?,
110            args: parse_token_expr(input, false)?,
111            has_bonus_display: false,
112            implied_bounds: Set::new(),
113        };
114        if attrs.display.is_some() {
115            return Err(Error::new_spanned(
116                attr,
117                "only one #[error(...)] attribute is allowed",
118            ));
119        }
120        attrs.display = Some(display);
121        Ok(())
122    })
123}

125fn parse_token_expr(input: ParseStream, mut begin_expr: bool) -> Result<TokenStream> {
126    let mut tokens = Vec::new();
127    while !input.is_empty() {
128        if begin_expr && input.peek(Token![.]) {
129            if input.peek2(Ident) {
130                input.parse::<Token![.]>()?;
131                begin_expr = false;
132                continue;
133            }
134            if input.peek2(LitInt) {
135                input.parse::<Token![.]>()?;
136                let int: Index = input.parse()?;
137                let ident = format_ident!("_{}", int.index, span = int.span);
138                tokens.push(TokenTree::Ident(ident));
139                begin_expr = false;
140                continue;
141            }
142        }
143
144        begin_expr = input.peek(Token![break])
145            || input.peek(Token![continue])
146            || input.peek(Token![if])
147            || input.peek(Token![in])
148            || input.peek(Token![match])
149            || input.peek(Token![mut])
150            || input.peek(Token![return])
151            || input.peek(Token![while])
152            || input.peek(Token![+])
153            || input.peek(Token![&])
154            || input.peek(Token![!])
155            || input.peek(Token![^])
156            || input.peek(Token![,])
157            || input.peek(Token![/])
158            || input.peek(Token![=])
159            || input.peek(Token![>])
160            || input.peek(Token![<])
161            || input.peek(Token![|])
162            || input.peek(Token![%])
163            || input.peek(Token![;])
164            || input.peek(Token![*])
165            || input.peek(Token![-]);
166
167        let token: TokenTree = if input.peek(token::Paren) {
168            let content;
169            let delimiter = parenthesized!(content in input);
170            let nested = parse_token_expr(&content, true)?;
171            let mut group = Group::new(Delimiter::Parenthesis, nested);
172            group.set_span(delimiter.span.join());
173            TokenTree::Group(group)
174        } else if input.peek(token::Brace) {
175            let content;
176            let delimiter = braced!(content in input);
177            let nested = parse_token_expr(&content, true)?;
178            let mut group = Group::new(Delimiter::Brace, nested);
179            group.set_span(delimiter.span.join());
180            TokenTree::Group(group)
181        } else if input.peek(token::Bracket) {
182            let content;
183            let delimiter = bracketed!(content in input);
184            let nested = parse_token_expr(&content, true)?;
185            let mut group = Group::new(Delimiter::Bracket, nested);
186            group.set_span(delimiter.span.join());
187            TokenTree::Group(group)
188        } else {
189            input.parse()?
190        };
191        tokens.push(token);
192    }
193    Ok(TokenStream::from_iter(tokens))
194}

196fn require_empty_attribute(attr: &Attribute) -> Result<()> {
197    let error_span = match &attr.meta {
198        Meta::Path(_) => return Ok(()),
199        Meta::List(meta) => meta.delimiter.span().open(),
200        Meta::NameValue(meta) => meta.eq_token.span,
201    };
202    Err(Error::new(
203        error_span,
204        "unexpected token in thiserror attribute",
205    ))
206}

208impl ToTokens for Display<'_> {
209    fn to_tokens(&self, tokens: &mut TokenStream) {
210        let fmt = &self.fmt;
211        let args = &self.args;
212        tokens.extend(quote! {
213            write!(__formatter, #fmt #args)
214        });
215    }
216}

218impl ToTokens for Trait {
219    fn to_tokens(&self, tokens: &mut TokenStream) {
220        let trait_name = format_ident!("{}", format!("{:?}", self));
221        tokens.extend(quote!(std::fmt::#trait_name));
222    }
