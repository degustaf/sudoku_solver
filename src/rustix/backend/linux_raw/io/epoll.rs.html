1//! epoll support.
2//!
3//! This is an experiment, and it isn't yet clear whether epoll is the right
4//! level of abstraction at which to introduce safety. But it works fairly well
5//! in simple examples ðŸ™‚.
6//!
7//! # Examples
8//!
9//! ```no_run
10//! # #![cfg_attr(io_lifetimes_use_std, feature(io_safety))]
11//! # #[cfg(feature = "net")]
12//! # fn main() -> std::io::Result<()> {
13//! use io_lifetimes::AsFd;
14//! use rustix::io::{epoll, ioctl_fionbio, read, write};
15//! use rustix::net::{
16//!     accept, bind_v4, listen, socket, AddressFamily, Ipv4Addr, Protocol, SocketAddrV4,
17//!     SocketType,
18//! };
19//! use std::collections::HashMap;
20//! use std::os::unix::io::AsRawFd;
21//!
22//! // Create a socket and listen on it.
23//! let listen_sock = socket(AddressFamily::INET, SocketType::STREAM, Protocol::default())?;
24//! bind_v4(&listen_sock, &SocketAddrV4::new(Ipv4Addr::LOCALHOST, 0))?;
25//! listen(&listen_sock, 1)?;
26//!
27//! // Create an epoll object. Using `Owning` here means the epoll object will
28//! // take ownership of the file descriptors registered with it.
29//! let epoll = epoll::epoll_create(epoll::CreateFlags::CLOEXEC)?;
30//!
31//! // Register the socket with the epoll object.
32//! epoll::epoll_add(&epoll, &listen_sock, 1, epoll::EventFlags::IN)?;
33//!
34//! // Keep track of the sockets we've opened.
35//! let mut next_id = 2;
36//! let mut sockets = HashMap::new();
37//!
38//! // Process events.
39//! let mut event_list = epoll::EventVec::with_capacity(4);
40//! loop {
41//!     epoll::epoll_wait(&epoll, &mut event_list, -1)?;
42//!     for (_event_flags, target) in &event_list {
43//!         if target == 1 {
44//!             // Accept a new connection, set it to non-blocking, and
45//!             // register to be notified when it's ready to write to.
46//!             let conn_sock = accept(&listen_sock)?;
47//!             ioctl_fionbio(&conn_sock, true)?;
48//!             epoll::epoll_add(
49//!                 &epoll,
50//!                 &conn_sock,
51//!                 next_id,
52//!                 epoll::EventFlags::OUT | epoll::EventFlags::ET,
53//!             )?;
54//!
55//!             // Keep track of the socket.
56//!             sockets.insert(next_id, conn_sock);
57//!             next_id += 1;
58//!         } else {
59//!             // Write a message to the stream and then unregister it.
60//!             let target = sockets.remove(&target).unwrap();
61//!             write(&target, b"hello\n")?;
62//!             let _ = epoll::epoll_del(&epoll, &target)?;
63//!         }
64//!     }
65//! }
66//! # }
67//! # #[cfg(not(feature = "net"))]
68//! # fn main() {}
69//! ```

149/// `epoll_ctl(self, EPOLL_CTL_ADD, data, event)`â€”Adds an element to an
150/// `Epoll`.
151///
152/// This registers interest in any of the events set in `events` occurring
153/// on the file descriptor associated with `data`.
154///
155/// Note that if `epoll_del` is not called on the I/O source passed into
156/// this function before the I/O source is `close`d, then the `epoll` will
157/// act as if the I/O source is still registered with it. This can lead to
158/// spurious events being returned from `epoll_wait`. If a file descriptor
159/// is an `Arc<dyn SystemResource>`, then `epoll` can be thought to maintain
160/// a `Weak<dyn SystemResource>` to the file descriptor.
161#[doc(alias = "epoll_ctl")]
162pub fn epoll_add(
163    epoll: impl AsFd,
164    source: impl AsFd,
165    data: u64,
166    event_flags: EventFlags,
167) -> io::Result<()> {
168    // SAFETY: We're calling `epoll_ctl` via FFI and we know how it
169    // behaves.
170    unsafe {
171        syscalls::epoll_add(
172            epoll.as_fd(),
173            source.as_fd().as_raw_fd(),
174            &linux_raw_sys::general::epoll_event {
175                events: event_flags.bits(),
176                data,
177            },
178        )
179    }
180}

182/// `epoll_ctl(self, EPOLL_CTL_MOD, target, event)`â€”Modifies an element in
183/// this `Epoll`.
184///
185/// This sets the events of interest with `target` to `events`.
186#[doc(alias = "epoll_ctl")]
187pub fn epoll_mod(
188    epoll: impl AsFd,
189    source: impl AsFd,
190    data: u64,
191    event_flags: EventFlags,
192) -> io::Result<()> {
193    // SAFETY: We're calling `epoll_ctl` via FFI and we know how it
194    // behaves.
195    unsafe {
196        let raw_fd = source.as_fd().as_raw_fd();
197        syscalls::epoll_mod(
198            epoll.as_fd(),
199            raw_fd,
200            &linux_raw_sys::general::epoll_event {
201                events: event_flags.bits(),
202                data,
203            },
204        )
205    }
206}

208/// `epoll_ctl(self, EPOLL_CTL_DEL, target, NULL)`â€”Removes an element in
209/// this `Epoll`.
210///
211/// This also returns the owning `Data`.
212#[doc(alias = "epoll_ctl")]
213pub fn epoll_del(epoll: impl AsFd, source: impl AsFd) -> io::Result<()> {
214    // SAFETY: We're calling `epoll_ctl` via FFI and we know how it
215    // behaves.
216    unsafe {
217        let raw_fd = source.as_fd().as_raw_fd();
218        syscalls::epoll_del(epoll.as_fd(), raw_fd)
219    }
220}

222/// `epoll_wait(self, events, timeout)`â€”Waits for registered events of
223/// interest.
224///
225/// For each event of interest, an element is written to `events`. On
226/// success, this returns the number of written elements.
227pub fn epoll_wait(
228    epoll: impl AsFd,
229    event_list: &mut EventVec,
230    timeout: c::c_int,
231) -> io::Result<()> {
232    // SAFETY: We're calling `epoll_wait` via FFI and we know how it
233    // behaves.
234    unsafe {
235        event_list.events.set_len(0);
236        let nfds = syscalls::epoll_wait(
237            epoll.as_fd(),
238            event_list.events[..].as_mut_ptr().cast(),
239            event_list.events.capacity(),
240            timeout,
241        )?;
242        event_list.events.set_len(nfds);
243    }
244
245    Ok(())
246}

248/// An iterator over the `Event`s in an `EventVec`.
249pub struct Iter<'a> {
250    iter: core::slice::Iter<'a, Event>,
251}
252
253impl<'a> Iterator for Iter<'a> {
254    type Item = (EventFlags, u64);
255
256    fn next(&mut self) -> Option<Self::Item> {
257        self.iter
258            .next()
259            .map(|event| (event.event_flags, event.data))
260    }
261}

263/// A record of an event that occurred.
264#[repr(C)]
265#[cfg_attr(target_arch = "x86_64", repr(packed))]
266struct Event {
267    // Match the layout of `linux_raw_sys::general::epoll_event`. We just use a
268    // `u64` instead of the full union.
269    event_flags: EventFlags,
270    data: u64,
271}
272
273/// A vector of `Event`s, plus context for interpreting them.
274pub struct EventVec {
275    events: Vec<Event>,
276}
277
278impl EventVec {
279    /// Constructs an `EventVec` with memory for `capacity` `Event`s.
280    #[inline]
281    pub fn with_capacity(capacity: usize) -> Self {
282        Self {
283            events: Vec::with_capacity(capacity),
284        }
285    }
286
287    /// Returns the current `Event` capacity of this `EventVec`.
288    #[inline]
289    pub fn capacity(&self) -> usize {
290        self.events.capacity()
291    }
292
293    /// Reserves enough memory for at least `additional` more `Event`s.
294    #[inline]
295    pub fn reserve(&mut self, additional: usize) {
296        self.events.reserve(additional);
297    }
298
299    /// Reserves enough memory for exactly `additional` more `Event`s.
300    #[inline]
301    pub fn reserve_exact(&mut self, additional: usize) {
302        self.events.reserve_exact(additional);
303    }
304
305    /// Clears all the `Events` out of this `EventVec`.
306    #[inline]
307    pub fn clear(&mut self) {
308        self.events.clear();
309    }
310
311    /// Shrinks the capacity of this `EventVec` as much as possible.
312    #[inline]
313    pub fn shrink_to_fit(&mut self) {
314        self.events.shrink_to_fit();
315    }
316
317    /// Returns an iterator over the `Event`s in this `EventVec`.
318    #[inline]
319    pub fn iter(&self) -> Iter<'_> {
320        Iter {
321            iter: self.events.iter(),
322        }
323    }
324
325    /// Returns the number of `Event`s logically contained in this `EventVec`.
326    #[inline]
327    pub fn len(&mut self) -> usize {
328        self.events.len()
329    }
330
331    /// Tests whether this `EventVec` is logically empty.
332    #[inline]
333    pub fn is_empty(&mut self) -> bool {
334        self.events.is_empty()
335    }
336}
337
338impl<'a> IntoIterator for &'a EventVec {
339    type IntoIter = Iter<'a>;
340    type Item = (EventFlags, u64);
341
342    #[inline]
343    fn into_iter(self) -> Self::IntoIter {
344        self.iter()
345    }
