96/// A buffer type used with `vmsplice`.
97/// It is guaranteed to be ABI compatible with the iovec type on Unix platforms
98/// and `WSABUF` on Windows. Unlike `IoSlice` and `IoSliceMut` it is
99/// semantically like a raw pointer, and therefore can be shared or mutated as
100/// needed.
101#[repr(transparent)]
102pub struct IoSliceRaw<'a> {
103    _buf: c::iovec,
104    _lifetime: PhantomData<&'a ()>,
105}
106
107impl<'a> IoSliceRaw<'a> {
108    /// Creates a new `IoSlice` wrapping a byte slice.
109    pub fn from_slice(buf: &'a [u8]) -> Self {
110        IoSliceRaw {
111            _buf: c::iovec {
112                iov_base: buf.as_ptr() as *mut u8 as *mut c::c_void,
113                iov_len: buf.len() as _,
114            },
115            _lifetime: PhantomData,
116        }
117    }
118
119    /// Creates a new `IoSlice` wrapping a mutable byte slice.
120    pub fn from_slice_mut(buf: &'a mut [u8]) -> Self {
121        IoSliceRaw {
122            _buf: c::iovec {
123                iov_base: buf.as_mut_ptr() as *mut c::c_void,
124                iov_len: buf.len() as _,
125            },
126            _lifetime: PhantomData,
127        }
128    }
